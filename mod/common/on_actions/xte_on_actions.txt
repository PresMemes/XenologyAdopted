# Triggers when the game starts
# Scopes: this = global/noscope
on_game_start = {
 	events = {
 		xte_traits_ev.1000
 	}
}

# Triggers when a country is generated/created
# Scopes: this = country
on_game_start_country = {
 	events = {
 		xte_traits_ev.998
	}
}

# No scope, like on_game_start

	}
}

# A new leader is created

 	}
}

	}
}
